import pandas as pd

def read_energy():
    energy = (pd.read_excel("Energy Indicators.xls", 
                       sheetname="Energy", header=None, skiprows=18, parse_cols=[2, 3, 4, 5], skip_footer=38,
                       names=['Country', 'Energy Supply', 'Energy Supply per Capita', '% Renewable'],
                       na_values='...')
                .set_index('Country'))

    energy['Energy Supply'] *= 1000000
    newindex = energy.index.to_series().astype(str).str.replace(r"\d+", "")
    newindex = newindex.str.replace(r" \(.*", "")
    newindex = newindex.replace("Republic of Korea", "South Korea")
    newindex = newindex.replace("United States of America", "United States")
    newindex = newindex.replace("United Kingdom of Great Britain and Northern Ireland", "United Kingdom")
    newindex = newindex.replace("China, Hong Kong Special Administrative Region", "Hong Kong")
    energy.index = newindex
    return energy

def read_gdp():
    gdp = (pd.read_csv("world_bank.csv", skiprows=4).set_index('Country Name'))
    newindex = gdp.index.to_series().astype(str).str.replace("Korea, Rep.", "South Korea")
    newindex = newindex.str.replace("Iran, Islamic Rep.", "Iran")
    newindex = newindex.replace("Republic of Korea", "South Korea")
    newindex = newindex.replace("Hong Kong SAR, China", "Hong Kong")
    gdp.index = newindex
    return gdp

def read_scimen():
    scimen = (pd.read_excel("scimagojr-3.xlsx")).set_index("Country")
    return scimen

def answer_one():
    all = pd.merge(read_scimen(), read_gdp(), left_index=True, right_index=True)
    all = pd.merge(all, read_energy(), left_index=True, right_index=True)
    all = all[['Rank', 'Documents', 'Citable documents', 'Citations', 'Self-citations', 'Citations per document', 'H index', 'Energy Supply', 'Energy Supply per Capita', '% Renewable', '2006', '2007', '2008', '2009', '2010', '2011', '2012', '2013', '2014', '2015']][all['Rank'] <= 15]
    return all

def answer_twelve():
    renw = answer_one()['% Renewable']
    bins = pd.cut(renw, 5)
    res = renw.groupby([lambda x: ContinentDict[x], bins]).size()
    res.index.names = ['Continent', '% Renewable']
